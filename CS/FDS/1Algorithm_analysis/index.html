    int max_subsequence_sum(int a[], unsigned int n)
{
    int thisssum = 0;
    int maxsum = 0;
    int best_i = -1, best_j = -1;
    for(int j = 0; j < n; j++)
    {
        thissum += a[j];
        if(thissum > maxsum)
        {
            maxsum = thissum;
            best_i = i;
            best_j = j;
        }
        if(thissum < 0)
        {
            i = j + 1;
            thissum = 0
        }
    }
    return maxsum;
}
